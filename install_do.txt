// Centos 7

```sh

// Install 3proxy
yum install wget

curl -O https://raw.githubusercontent.com/duchuyta/proxy6/develop/install_do.sh
bash install_do.sh "eth0"

// Setup IPv6 
vi /etc/sysconfig/network-scripts/ifcfg-eth0

IPV6INIT=yes
IPV6ADDR=2604:a880:400:d0::1a08:0/64
IPV6_DEFAULTGW=2604:a880:400:d0::1
IPV6_AUTOCONF=no
DNS1=2001:4860:4860::8844
DNS2=2001:4860:4860::8888
DNS3=209.244.0.3

sudo systemctl restart network

// Update config
vi /usr/local/etc/3proxy/3proxy.cfg

// ifconfig
vi /root/proxy-installer/boot_ifconfig.sh

// iptables
vi /root/proxy-installer/boot_iptables.sh

// rc.local
vi /etc/rc.local

```
